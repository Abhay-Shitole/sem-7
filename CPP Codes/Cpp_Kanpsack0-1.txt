


#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int knapsackDP(int W, vector<int> wt, vector<int> val, int n) {
    vector<vector<int>> dp(n + 1, vector<int>(W + 1, 0));

    // Build DP table
    for (int i = 1; i <= n; i++) {
        for (int w = 1; w <= W; w++) {
            if (wt[i - 1] <= w)
                dp[i][w] = max(val[i - 1] + dp[i - 1][w - wt[i - 1]], dp[i - 1][w]);
            else
                dp[i][w] = dp[i - 1][w];
        }
    }

    // Print DP table (optional for understanding)
    cout << "\nDP Table:\n";
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            cout << dp[i][w] << "\t";
        }
        cout << endl;
    }

    return dp[n][W];
}

int main() {
    int n, W;
    cout << "Enter number of items: ";
    cin >> n;

    vector<int> val(n), wt(n);
    cout << "Enter value and weight of each item:\n";
    for (int i = 0; i < n; i++)
        cin >> val[i] >> wt[i];

    cout << "Enter capacity of knapsack: ";
    cin >> W;

    int maxValue = knapsackDP(W, wt, val, n);

    cout << "\nMaximum value in Knapsack = " << maxValue << endl;

    return 0;
}



/*0-1 KNAPSACK PROBLEM USING DYNAMIC PROGRAMMING
==============================================

THEORY
-------
Problem Statement:
You are given 'n' items, each with a value and weight, and a knapsack of capacity W.
The goal is to select items to maximize total value without exceeding capacity.
In 0-1 Knapsack, you cannot divide an item â€” it is either taken or not.

Difference between 0-1 and Fractional Knapsack:
------------------------------------------------
| Feature | 0-1 Knapsack | Fractional Knapsack |
|----------|---------------|--------------------|
| Division | Not allowed | Allowed |
| Method | Dynamic Programming / Branch & Bound | Greedy Method |
| Optimal Substructure | Yes | Yes |
| Overlapping Subproblems | Yes | No |
| Time Complexity | O(nW) | O(n log n) |

Dynamic Programming Approach:
------------------------------
We define dp[i][w] = maximum value using first i items and capacity w.

Recurrence Relation:
dp[i][w] = dp[i-1][w]                                   
 if weight[i] > w
    max(dp[i-1][w], value[i] + dp[i-1][w - weight[i]]) if weight[i] <= w

Base Cases:
dp[0][w] = 0, dp[i][0] = 0

Time Complexity: O(n * W)
Space Complexity: O(n * W)*/
